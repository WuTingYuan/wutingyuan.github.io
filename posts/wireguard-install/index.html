RotuerOS Wireguard 安裝

2023-01-14 — Written by 地瓜酥胡了
#RouterOS  #Wireguard 

架構 #

RouterOS # 

# 建立 Wiregaurd 介面，Public Key（自動產生）
/interface wireguard add listen-port=13231 mtu=1420 name=wireguard1

# Firewall 開 Input Port
/ip firewall filter add action=accept chain=input comment=WireGuard dst-port=13231 protocol=udp

# Firewall 開 VPN 網段到內網（雙向）
/ip firewall filter add action=accept chain=forward dst-address=192.168.1.0/24 src-address=10.10.10.0/24
/ip firewall filter add action=accept chain=forward dst-address=10.10.10.0/24 src-address=192.168.1.0/24

# 建立 Wireguard IP Address （設定完 network 會自動產升）
/ip address add address=10.10.10.0/24 interface=wireguard1

# Client 設定完成後取得 Client 的 Public Key
/interface wireguard peers add allowed-address=10.10.10.1/32 interface=wireguard1 public-key="<Public Key>"
